\def\leftframe#1{
\node (lc) [frame, left=of #1, xshift=-2mm] {\frametext}
  edge [->] (#1.west);
}

\def\frameadjust{0.1}
\def\leftframes#1{
\node (lt) [frame, above left=of #1] {\frametext}
  edge [->, bend right] ($(#1.west |- #1.north) - (0,\frameadjust)$);
\node (lb) [frame, below left=of #1] {\frametext}
  edge [->, bend left] ($(#1.west |- #1.south) + (0,\frameadjust)$);
\leftframe{#1}
}

\def\rightframe#1{
\node (rc) [frame, right=of #1, xshift=2mm] {\frametext}
  edge [<-] (#1.east);
}

\def\rightframes#1{
\node (rt) [frame, above right=of #1] {\frametext}
  edge [<-, bend left] ($(#1.east |- #1.north) - (0,\frameadjust)$);
\node (rb) [frame, below right=of #1] {\frametext}
  edge [<-, bend right] ($(#1.east |- #1.south) + (0,\frameadjust)$);
\rightframe{#1}
}
